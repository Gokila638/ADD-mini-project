<script>
function startEffect(event){
    let form = document.querySelector("form");

    // ✅ Check form
    if(!form.checkValidity()){
        alert("⚠ Please fill all required details!");
        return;
    }

    event.preventDefault();

    // 🔊 SOUND FIX
    let sound = document.getElementById("sound");
    sound.pause();
    sound.currentTime = 0;

    let playPromise = sound.play();
    if (playPromise !== undefined) {
        playPromise.catch(() => {
            console.log("Sound blocked by browser");
        });
    }

    // ⭐ STARS FROM LEFT CORNER
    for(let i=0;i<50;i++){
        let star = document.createElement("div");
        star.innerHTML = "★";

        star.style.position = "fixed";
        star.style.color = "gold";

        // ⭐ Random size (small + big mix)
        let size = Math.random() * 20 + 10;
        star.style.fontSize = size + "px";

        // ⭐ Start from LEFT BOTTOM
        star.style.left = "0px";
        star.style.top = window.innerHeight + "px";

        star.style.zIndex = "9999";

        document.body.appendChild(star);

        // ⭐ Move diagonally
        let moveX = Math.random() * window.innerWidth;
        let moveY = -Math.random() * 500 - 200;

        star.animate([
            { transform: "translate(0,0)", opacity:1 },
            { transform: `translate(${moveX}px, ${moveY}px)`, opacity:0 }
        ],{
            duration: 1500,
            easing: "ease-out"
        });

        setTimeout(()=>star.remove(),1500);
    }

    // ⏳ Redirect after animation
    setTimeout(()=>{
        window.location.href = "payment.html";
    },1500);
}
</script>
